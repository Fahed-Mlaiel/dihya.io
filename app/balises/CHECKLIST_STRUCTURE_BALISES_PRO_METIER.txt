# CHECKLIST_STRUCTURE_BALISES_PRO_METIER.txt

# Structure professionnelle et exhaustive du dossier /balises pour un projet multi-stack, industrialisable, clé en main
# (Hors tests, selon cahier des charges, état réel et bonnes pratiques)

# Dossier principal
balises/

# Fichiers recommandés à la racine de balises
README.md               # Documentation sur la gestion des balises (SEO, accessibilité, OpenGraph, microdata, etc.), conventions, usages
index.js                # Export centralisé des balises pour import facilité dans le code (React, Node.js)
__init__.py             # (optionnel, si usage Python : permet l'import en tant que module)

# Sous-dossiers types
seo/                    # Balises SEO (title, meta, canonical, etc.)
open_graph/             # Balises OpenGraph (partage réseaux sociaux)
microdata/              # Microdonnées (schema.org, rich snippets)
accessibility/          # Balises ARIA, accessibilité, navigation clavier
custom/                 # Balises personnalisées, extensions, plugins

# Exemples de fichiers métiers à placer dans chaque sous-dossier
seo/
    title.template.js           # Template de balise <title>
    meta.template.js            # Template de balises <meta>
    canonical.template.js       # Template de balise <link rel="canonical">
    seoConfig.json              # Config SEO globale
    README.md                   # Documentation SEO

open_graph/
    og_tags.template.js         # Template de balises OpenGraph
    ogConfig.json               # Config OpenGraph globale
    README.md                   # Documentation OpenGraph

microdata/
    schemaOrg.template.js       # Template schema.org
    richSnippets.template.js    # Template rich snippets
    microdataConfig.json        # Config microdata globale
    README.md                   # Documentation microdata

accessibility/
    aria.template.js            # Template balises ARIA
    navigation.template.js      # Template navigation clavier
    accessibilityConfig.json    # Config accessibilité globale
    README.md                   # Documentation accessibilité

custom/
    customTag.template.js       # Template balise personnalisée
    pluginTag.template.js       # Template balise plugin
    customConfig.json           # Config custom globale
    README.md                   # Documentation custom

# Bonnes pratiques :
# - Centraliser la documentation et les templates pour chaque type de balise
# - Respecter les standards SEO, accessibilité, OpenGraph, schema.org
# - Faciliter l’import et la réutilisation dans le code (index.js)
# - Adapter la structure selon les besoins métiers et stacks utilisées
